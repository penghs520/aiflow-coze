## 1. 概述

### 1.1 安全目标
构建全方位、多层次的安全防护体系，确保AI工作流平台在与扣子工作流集成过程中的安全性，包括：

1. **数据安全**：保护用户数据在传输、处理、存储各环节的机密性和完整性
2. **访问控制**：确保只有授权用户和系统可以访问相应资源
3. **系统防护**：防范各类网络攻击和恶意行为
4. **合规保障**：满足相关法律法规和行业标准要求

### 1.2 安全原则
1. **最小权限原则**：每个组件、用户、进程只拥有完成其任务所必需的最小权限
2. **纵深防御原则**：构建多层安全防护，单一防御层失效不影响整体安全
3. **默认拒绝原则**：未明确允许的访问一律拒绝
4. **持续监控原则**：实时监控安全状态，及时发现和响应安全事件

```java
@Component
public class PasswordSecurityService {
    
    private static final int BCRYPT_STRENGTH = 12;
    
    public String hashPassword(String rawPassword) {
        return BCrypt.hashpw(rawPassword, BCrypt.gensalt(BCRYPT_STRENGTH));
    }
    
    public boolean verifyPassword(String rawPassword, String hashedPassword) {
        return BCrypt.checkpw(rawPassword, hashedPassword);
    }
    
    // 密码强度检查
    public PasswordStrength checkPasswordStrength(String password) {
        PasswordStrength strength = new PasswordStrength();
        
        // 长度要求
        strength.setLengthOk(password.length() >= 8);
        
        // 复杂度要求
        strength.setHasUppercase(password.matches(".*[A-Z].*"));
        strength.setHasLowercase(password.matches(".*[a-z].*"));
        strength.setHasDigit(password.matches(".*\\d.*"));
        strength.setHasSpecial(password.matches(".*[!@#$%^&*()].*"));
        
        // 常见弱密码检查
        strength.setCommonPassword(!isCommonPassword(password));
        
        return strength;
    }
}
```

```java
public class ResourceAccessControl {
    
    @Resource
    private PermissionService permissionService;
    
    /**
     * 检查用户对特定资源的访问权限
     */
    public boolean checkResourceAccess(String userId, 
                                      ResourceType resourceType, 
                                      String resourceId, 
                                      Action action) {
        
        // 1. 获取用户所有角色
        List<Role> roles = roleService.getUserRoles(userId);
        
        // 2. 检查每个角色的权限
        for (Role role : roles) {
            if (permissionService.hasPermission(role.getId(), 
                                               resourceType, 
                                               resourceId, 
                                               action)) {
                return true;
            }
        }
        
        // 3. 检查默认权限
        return checkDefaultPermission(resourceType, resourceId, action);
    }
    
    /**
     * 工作流访问控制
     */
    public boolean canAccessWorkflow(String userId, String workflowId) {
        // 检查工作流状态
        Workflow workflow = workflowService.getWorkflow(workflowId);
        
        if (!workflow.isPublished()) {
            return false; // 未发布的工作流无法访问
        }
        
        // 检查用户权限
        return checkResourceAccess(userId, 
                                  ResourceType.WORKFLOW, 
                                  workflowId, 
                                  Action.EXECUTE);
    }
}
```

```java
@Component
public class RequestSigner {
    
    private static final String HMAC_ALGORITHM = "HmacSHA256";
    private static final long TIMESTAMP_TOLERANCE = 5 * 60 * 1000; // 5分钟
    
    /**
     * 生成请求签名
     */
    public String generateSignature(String method, 
                                    String path, 
                                    String body, 
                                    long timestamp) {
        
        // 构建签名字符串
        String signString = String.format("%s\n%s\n%s\n%d",
            method.toUpperCase(),
            path,
            Optional.ofNullable(body).orElse(""),
            timestamp
        );
        
        // 计算HMAC
        Mac mac = Mac.getInstance(HMAC_ALGORITHM);
        SecretKeySpec keySpec = new SecretKeySpec(
            getSecretKey().getBytes(StandardCharsets.UTF_8), 
            HMAC_ALGORITHM
        );
        mac.init(keySpec);
        
        byte[] signature = mac.doFinal(signString.getBytes(StandardCharsets.UTF_8));
        return Hex.encodeHexString(signature);
    }
    
    /**
     * 验证请求签名
     */
    public boolean verifySignature(HttpServletRequest request, String body) {
        // 获取请求头
        String signature = request.getHeader("X-Signature");
        String timestampStr = request.getHeader("X-Timestamp");
        
        if (signature == null || timestampStr == null) {
            return false;
        }
        
        // 验证时间戳
        long timestamp = Long.parseLong(timestampStr);
        long currentTime = System.currentTimeMillis();
        
        if (Math.abs(currentTime - timestamp) > TIMESTAMP_TOLERANCE) {
            return false; // 时间戳超时
        }
        
        // 计算预期签名
        String method = request.getMethod();
        String path = request.getRequestURI();
        String query = request.getQueryString();
        if (query != null) {
            path += "?" + query;
        }
        
        String expectedSignature = generateSignature(method, path, body, timestamp);
        
        // 安全比较签名（避免时序攻击）
        return MessageDigest.isEqual(
            signature.getBytes(StandardCharsets.UTF_8),
            expectedSignature.getBytes(StandardCharsets.UTF_8)
        );
    }
}
```

```java
     * 加密存储文件

    public EncryptedFile encryptFile(File file, String userId) {

        SecretKey fileKey = keyManagementService.generateFileKey();

        Cipher cipher = Cipher.getInstance("AES/GCM/NoPadding");

            .encryptedKey(encryptedKey)
            .iv(cipher.getIV())
            .authTag(extractAuthTag(cipher))
            .build();
    }
```

```java
@Component
public class DDoSProtectionService {
    
    @Resource
    private RedisTemplate<String, String> redisTemplate;
    
    private static final String RATE_LIMIT_KEY = "ddos:rate:{}:{}"; // {userId}:{ip}
    private static final int WINDOW_SECONDS = 60;
    private static final int MAX_REQUESTS = 100;
    
    /**
     * 检查是否超过速率限制
     */
    public boolean isRateLimitExceeded(String userId, String ip) {
        String key = String.format(RATE_LIMIT_KEY, userId, ip);
        
        // 滑动窗口计数
        Long current = redisTemplate.opsForValue().increment(key);
        if (current == 1) {
            // 第一次请求，设置过期时间
            redisTemplate.expire(key, WINDOW_SECONDS, TimeUnit.SECONDS);
        }
        
        return current > MAX_REQUESTS;
    }
    
    /**
     * IP黑名单管理
     */
    public boolean isIpBlacklisted(String ip) {
        String key = "security:ip:blacklist";
        return redisTemplate.opsForSet().isMember(key, ip);
    }
    
    /**
     * 可疑行为检测
     */
    public SuspiciousBehavior detectSuspiciousBehavior(HttpServletRequest request) {
        SuspiciousBehavior behavior = new SuspiciousBehavior();
        
        // 检查User-Agent
        String userAgent = request.getHeader("User-Agent");
        if (isSuspiciousUserAgent(userAgent)) {
            behavior.setSuspiciousUserAgent(true);
        }
        
        // 检查请求频率
        String ip = request.getRemoteAddr();
        if (isHighFrequencyRequest(ip)) {
            behavior.setHighFrequency(true);
        }
        
        // 检查请求参数异常
        if (hasSuspiciousParameters(request)) {
            behavior.setSuspiciousParameters(true);
        }
        
        return behavior;
    }
}
```

```java
@Component
@Validated
public class InputValidationService {
    
    /**
     * 工作流参数验证
     */
    public ValidationResult validateWorkflowParameters(
            String workflowId, 
            @Valid @NotNull Map<String, Object> parameters) {
        
        ValidationResult result = new ValidationResult();
        
        // 1. 获取工作流验证规则
        Workflow workflow = workflowService.getWorkflow(workflowId);
        ValidationRules rules = workflow.getValidationRules();
        
        // 2. 验证必填字段
        for (String requiredField : rules.getRequiredFields()) {
            if (!parameters.containsKey(requiredField)) {
                result.addError(requiredField, "必填字段缺失");
            }
        }
        
        // 3. 验证数据类型和范围
        for (Map.Entry<String, Object> entry : parameters.entrySet()) {
            String fieldName = entry.getKey();
            Object value = entry.getValue();
            
            // 检查数据类型
            ParameterType expectedType = rules.getParameterType(fieldName);
            if (!isValidType(value, expectedType)) {
                result.addError(fieldName, 
                    String.format("数据类型错误，期望：%s", expectedType));
            }
            
            // 检查值范围
            if (!isInRange(value, rules.getRange(fieldName))) {
                result.addError(fieldName, "值超出允许范围");
            }
            
            // 检查安全字符
            if (value instanceof String && containsDangerousChars((String) value)) {
                result.addError(fieldName, "包含危险字符");
            }
        }
        
        // 4. 文件上传验证
        if (parameters.containsKey("file")) {
            FileUploadParam fileParam = parseFileParam(parameters.get("file"));
            if (!validateFileUpload(fileParam)) {
                result.addError("file", "文件上传验证失败");
            }
        }
        
        return result;
    }
    
    /**
     * 文件上传安全验证
     */
    public boolean validateFileUpload(FileUploadParam fileParam) {
        // 检查文件类型
        if (!isAllowedFileType(fileParam.getFileType())) {
            return false;
        }
        
        // 检查文件大小
        if (fileParam.getFileSize() > MAX_FILE_SIZE) {
            return false;
        }
        
        // 检查文件内容
        if (!isSafeFileContent(fileParam.getFileContent())) {
            return false;
        }
        
        // 检查文件名安全
        if (containsDangerousChars(fileParam.getFileName())) {
            return false;
        }
        
        return true;
    }
}
```

```java
@Configuration
public class JwtSecurityConfig {
    
    @Value("${jwt.secret.key}")
    private String secretKey;
    
    @Value("${jwt.expiration.time}")
    private long expirationTime;
    
    @Bean
    public JwtDecoder jwtDecoder() {
        SecretKey key = Keys.hmacShaKeyFor(secretKey.getBytes(StandardCharsets.UTF_8));
        
        return NimbusJwtDecoder.withSecretKey(key)
            .macAlgorithm(MacAlgorithm.HS256)
            .build();
    }
    
    @Bean
    public JwtEncoder jwtEncoder() {
        SecretKey key = Keys.hmacShaKeyFor(secretKey.getBytes(StandardCharsets.UTF_8));
        
        return new NimbusJwtEncoder(
            new ImmutableSecret<>(key)
        );
    }
    
    @Bean
    public JwtSecurityFilter jwtFilter() {
        return new JwtSecurityFilter(
            jwtDecoder(),
            userDetailsService()
        );
    }
    
    public String generateToken(UserDetails userDetails) {
        Instant now = Instant.now();
        Instant expiry = now.plusMillis(expirationTime);
        
        JwtClaimsSet claims = JwtClaimsSet.builder()
            .issuer("ai-workflow-platform")
            .issuedAt(now)
            .expiresAt(expiry)
            .subject(userDetails.getUsername())
            .claim("authorities", 
                userDetails.getAuthorities().stream()
                    .map(GrantedAuthority::getAuthority)
                    .collect(Collectors.toList()))
            .build();
        
        return jwtEncoder().encode(JwtEncoderParameters.from(claims)).getTokenValue();
    }
}
```

```java
@Component
public class SessionSecurityManager {
    
    @Resource
    private RedisTemplate<String, String> redisTemplate;
    
    private static final String SESSION_PREFIX = "session:";
    private static final long MAX_SESSION_AGE = 24 * 60 * 60 * 1000; // 24小时
    private static final int MAX_CONCURRENT_SESSIONS = 5;
    
    /**
     * 创建安全会话
     */
    public SecureSession createSession(String userId, HttpServletRequest request) {
        // 检查并发会话数
        long sessionCount = redisTemplate.opsForList()
            .size(SESSION_PREFIX + userId);
        
        if (sessionCount >= MAX_CONCURRENT_SESSIONS) {
            // 终止最老的会话
            terminateOldestSession(userId);
        }
        
        // 生成会话信息
        SecureSession session = SecureSession.builder()
            .sessionId(UUID.randomUUID().toString())
            .userId(userId)
            .ipAddress(request.getRemoteAddr())
            .userAgent(request.getHeader("User-Agent"))
            .createdAt(System.currentTimeMillis())
            .lastAccessed(System.currentTimeMillis())
            .isActive(true)
            .build();
        
        // 存储会话
        String sessionKey = SESSION_PREFIX + userId + ":" + session.getSessionId();
        redisTemplate.opsForValue().set(
            sessionKey,
            JsonUtils.toJson(session),
            MAX_SESSION_AGE,
            TimeUnit.MILLISECONDS
        );
        
        // 记录会话列表
        redisTemplate.opsForList().rightPush(
            SESSION_PREFIX + userId,
            session.getSessionId()
        );
        
        return session;
    }
    
    /**
     * 验证会话安全性
     */
    public SessionSecurity validateSessionSecurity(String sessionId, 
                                                  HttpServletRequest request) {
        
        SessionSecurity security = new SessionSecurity();
        
        // 检查IP变化
        String currentIp = request.getRemoteAddr();
        String originalIp = getSessionIp(sessionId);
        
        if (!currentIp.equals(originalIp)) {
            security.setIpChanged(true);
            security.setRiskLevel(RiskLevel.MEDIUM);
            
            // 记录IP变更事件
            logSecurityEvent("SESSION_IP_CHANGE", 
                String.format("Session %s: %s -> %s", 
                    sessionId, originalIp, currentIp));
        }
        
        // 检查User-Agent变化
        String currentUserAgent = request.getHeader("User-Agent");
        String originalUserAgent = getSessionUserAgent(sessionId);
        
        if (!currentUserAgent.equals(originalUserAgent)) {
            security.setUserAgentChanged(true);
            security.setRiskLevel(RiskLevel.HIGH);
            
            // 终止会话
            terminateSession(sessionId);
        }
        
        // 检查会话闲置时间
        long lastAccessed = getSessionLastAccessed(sessionId);
        long idleTime = System.currentTimeMillis() - lastAccessed;
        
        if (idleTime > MAX_IDLE_TIME) {
            security.setIdleTimeout(true);
            terminateSession(sessionId);
        }
        
        return security;
    }
}
```

```java
@Component
public class IntrusionDetectionSystem {
    
    @Resource
    private SecurityEventCollector eventCollector;
    
    @Resource
    private AlertService alertService;
    
    /**
     * 实时入侵检测
     */
    @Scheduled(fixedDelay = 10000) // 每10秒检测一次
    public void detectIntrusions() {
        // 收集最近的安全事件
        List<SecurityEvent> recentEvents = eventCollector
            .collectRecentEvents(60000); // 最近60秒
        
        // 检测异常模式
        List<IntrusionAlert> alerts = new ArrayList<>();
        
        // 1. 暴力破解检测
        alerts.addAll(detectBruteForce(recentEvents));
        
        // 2. SQL注入尝试
        alerts.addAll(detectSqlInjection(recentEvents));
        
        // 3. 恶意文件上传
        alerts.addAll(detectMaliciousUpload(recentEvents));
        
        // 4. 敏感数据访问
        alerts.addAll(detectSensitiveDataAccess(recentEvents));
        
        // 发送告警
        for (IntrusionAlert alert : alerts) {
            handleIntrusionAlert(alert);
        }
    }
    
    private List<IntrusionAlert> detectBruteForce(List<SecurityEvent> events) {
        List<IntrusionAlert> alerts = new ArrayList<>();
        
        // 按IP分组统计登录失败
        Map<String, Integer> failedAttempts = events.stream()
            .filter(e -> e.getEventType() == SecurityEventType.LOGIN_FAILURE)
            .collect(Collectors.groupingBy(
                SecurityEvent::getIpAddress,
                Collectors.collectingAndThen(
                    Collectors.counting(),
                    Long::intValue
                )
            ));
        
        // 检测异常
        for (Map.Entry<String, Integer> entry : failedAttempts.entrySet()) {
            if (entry.getValue() > 10) { // 10分钟内失败10次以上
                IntrusionAlert alert = IntrusionAlert.builder()
                    .alertType(AlertType.BRUTE_FORCE)
                    .ipAddress(entry.getKey())
                    .severity(Severity.HIGH)
                    .description(String.format("暴力破解尝试，失败次数: %d", entry.getValue()))
                    .timestamp(System.currentTimeMillis())
                    .build();
                
                alerts.add(alert);
            }
        }
        
        return alerts;
    }
}
```

### 8.1 安全事件响应流程
```java
@Component
public class SecurityIncidentResponse {
    
    @Resource
    private IncidentManagementSystem incidentSystem;
    
    @Resource
    private EmergencyContactService contactService;
    
    /**
     * 处理安全事件
     */
    public IncidentResponse handleSecurityIncident(SecurityIncident incident) {
        
        // 1. 事件分类与定级
        IncidentLevel level = classifyIncident(incident);
        
        // 2. 启动应急响应
        IncidentResponse response = IncidentResponse.builder()
            .incidentId(incident.getId())
            .startTime(System.currentTimeMillis())
            .status(IncidentStatus.ACTIVE)
            .level(level)
            .build();
        
        // 3. 根据严重级别采取行动
        switch (level) {
            case CRITICAL:
                // 紧急响应：隔离系统、通知管理层
                handleCriticalIncident(incident, response);
                break;
                
            case HIGH:
                // 高级响应：限制访问、详细调查
                handleHighSeverityIncident(incident, response);
                break;
                
            case MEDIUM:
                // 常规响应：修复问题、记录事件
                handleMediumSeverityIncident(incident, response);
                break;
                
            case LOW:
                // 基础响应：记录并观察
                handleLowSeverityIncident(incident, response);
                break;
        }
        
        // 4. 记录响应过程
        incidentSystem.recordResponse(response);
        
        return response;
    }
    
    private void handleCriticalIncident(SecurityIncident incident, 
                                       IncidentResponse response) {
        
        // 立即隔离受影响的系统
        isolateAffectedSystems(incident.getAffectedSystems());
        
        // 通知管理层和安全团队
        contactService.notifyEmergencyContacts(
            "CRITICAL_SECURITY_INCIDENT",
            String.format("严重安全事件: %s", incident.getDescription())
        );
        
        // 启动取证调查
        startForensicInvestigation(incident);
        
        // 准备公开声明（如果需要）
        preparePublicStatement(incident);
    }
}
```
